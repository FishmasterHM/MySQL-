聚集函数(aggregate fuction)
  
AVG()   返回某列的平均值
SELECT AVG(prod_price) AS avg_price
From products;

COUNT() 返回某列的行数
SELECT COUNT(*) AS num_cust
FROM customers；
计算全部行数(包括NULL)
       COUNT(name) 
不包括NULL

MAX()   返回某列的最大值
MIN()   返回某列的最小值
SUM()   返回某列值之和

DISTINCT 检索不同的值
SELECT AVG(DISTINCT prod_price) AS avg_price
FROM products
WHERE vend_id = 1003;
